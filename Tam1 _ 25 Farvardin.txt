#برنامه‌ای بنویسید که جدول ضرب اعداد ۳ تا ۵ را در خروجی چاپ کند. در جدول ضرب عدد ۳، عدد ۳ در همه اعداد ضرب شود به غیر از عدد ۳. در جدول ضرب عدد ۴، عدد ۴ در همه اعداد ضرب شود به غیر از عدد ۳ و همین طور ادامه بدهید.

for (let i = 3; i <= 5; i++) {
  console.log("جدول ضرب عدد " + i);
  for (let j = 1; j <= 10; j++) {
    if (j === 3) continue;
    console.log(i + " x " + j + " = " + (i * j));
  }
  console.log("--------------------------");
}
